# NOTE: dvorak (src/external/dvorak/dvorak.asm) is deliberately not wired up
# as a build target — it has known fundamental problems (see header comment

# Dynamic build numbers custom commands
# Re-runs the increment script whenever a dependency's mtime changes, but the
# script itself only bumps BUILD_OS when the sources' content hash actually
# differs from the last recorded one (see cmake/IncrementBuildNumber.cmake).
get_filename_component(CMD64_ENTRY_ABS "${CMD64_ENTRY}" ABSOLUTE)
set(OS_HASH_SOURCES "${CMD64_ENTRY_ABS}" ${CMD64_SRCS})
list(REMOVE_DUPLICATES OS_HASH_SOURCES)
list(SORT OS_HASH_SOURCES)
set(OS_HASH_SOURCES_FILE "${CMAKE_BINARY_DIR}/build_os_sources.txt")
string(REPLACE ";" "\n" OS_HASH_SOURCES_CONTENT "${OS_HASH_SOURCES}")
file(WRITE "${OS_HASH_SOURCES_FILE}" "${OS_HASH_SOURCES_CONTENT}\n")

add_custom_command(
    OUTPUT "${CMAKE_BINARY_DIR}/build_os.inc"
    COMMAND "${CMAKE_COMMAND}"
        -DBUILD_FILE="${CMAKE_SOURCE_DIR}/src/command64/BUILD_OS"
        -DINC_FILE="${CMAKE_BINARY_DIR}/build_os.inc"
        -DDEFAULT_VERSION=2417
        -DVAR_NAME="BUILD_NUMBER"
        -DSOURCES_LIST_FILE="${OS_HASH_SOURCES_FILE}"
        -P "${CMAKE_SOURCE_DIR}/cmake/IncrementBuildNumber.cmake"
    DEPENDS ${CMD64_SRCS} "${CMD64_ENTRY_ABS}"
    COMMENT "Checking/Incrementing OS build counter"
)

# 1. OS PRG Target

    DEPENDS "${CMAKE_BINARY_DIR}/build_os.inc" "${CMAKE_BINARY_DIR}/build_config_default/build_config.inc"
    INCLUDE_DIRS "${CMAKE_BINARY_DIR}/build_config_default" "${CMAKE_BINARY_DIR}"
)

# 2. DEBUG Utility PRG Target — built with ca65/ld65 after byte-for-byte
# parity verification against the previous KickAssembler build. Ships
# unconditionally in IMAGE_PRG_TARGETS (like LABEL/CONWAY below), so a
# missing ca65/ld65 toolchain fails the entire configure rather than
# silently skipping this target.
if(Ca65_FOUND)
    add_ca65_app(debug "${DEBUG_ENTRY}" DEBUG_SRCS 1012 "2000")
else()
    message(FATAL_ERROR "ca65/ld65 not found on PATH — required for DEBUG target")
endif()

# 3. LABEL External Command PRG Target — built with ca65/ld65 (migrated
# off the spike/ca65-label/ branch; see
# brain/plans/2026-07-08-ca65-adoption-and-spike-migration.md Phase 4).
# Ships unconditionally in IMAGE_PRG_TARGETS, so — like DEBUG — a missing
# ca65/ld65 toolchain fails the entire configure rather than silently
# skipping this target.
if(Ca65_FOUND)
    add_ca65_app(label "${LABEL_ENTRY}" LABEL_SRCS 1032 "0700")
else()
    message(FATAL_ERROR "ca65/ld65 not found on PATH — required for LABEL target")
endif()
set(LABEL_TARGET label)

# 4. CONWAY Game of Life — built with ca65/ld65 (migrated off the
# spike/ca65-conway/ branch; see
# brain/plans/2026-07-08-ca65-adoption-and-spike-migration.md Phase 4).
# Two object files (conway_main.s / conway_grid.s) linked together by
# ld65; CODE_ALIGN=256 because conway_grid.s embeds two page-aligned grid
# buffers directly in CODE. Ships unconditionally in IMAGE_PRG_TARGETS, so
# — like DEBUG/LABEL — a missing ca65/ld65 toolchain fails the entire
# configure rather than silently skipping this target.
if(Ca65_FOUND)
    add_ca65_app(conway "${CONWAY_ENTRY}" CONWAY_SRCS 1040 "0C00" "256")
else()
    message(FATAL_ERROR "ca65/ld65 not found on PATH — required for CONWAY target")
endif()
set(CONWAY_TARGET conway)

# 4f. ca65/ld65 test programs (tests/src/<name>/<name>.s, migrated off the
# spike branch; see
# brain/plans/2026-07-08-ca65-adoption-and-spike-migration.md Phase 5).
# Excludes reloc, whose entire premise is Kick's aptRelocate/reloc.py
# patch mechanism, which doesn't exist in ca65/ld65 output -- excluded
# implicitly, since tests/src/reloc/ has no .s file for the glob to match.
# Kept dual with the Kick tests/src/*/*.asm counterparts (not replaced):
# ongoing regression coverage for both toolchains, not a shipping
# duplicate to eliminate.
set(TESTS_CA65_TARGETS "")
if(Ca65_FOUND)
    file(GLOB CA65_SHARED_INCLUDES "include/ca65/*.inc")
    file(GLOB TESTS_CA65_SRCS "tests/src/*/*.s")
    foreach(TEST_CA65_SRC ${TESTS_CA65_SRCS})
        get_filename_component(TEST_CA65_NAME "${TEST_CA65_SRC}" NAME_WE)
        set(TEST_CA65_TARGET "test_ca65_${TEST_CA65_NAME}")
        set(TMP_CA65_SRCS_${TEST_CA65_NAME} "${CA65_SHARED_INCLUDES}")
        add_ca65_app(${TEST_CA65_TARGET} "${TEST_CA65_SRC}" TMP_CA65_SRCS_${TEST_CA65_NAME} 1000 "0700")
        list(APPEND TESTS_CA65_TARGETS ${TEST_CA65_TARGET})
    endforeach()
endif()

# 4g. Phase 3 smoke test for add_ca65_app (cmake/Ca65.cmake) -- proves the
# production ca65 build pipeline (BUILD_<NAME> versioning, hash-gated
# build-number bump, templated .cfg generation, -I include/ca65 + -t c64,
# base/next-page relocation build) end-to-end, independent of any real
# test/app target. Reuses tests/src/hello/hello.s (the Phase 5 migration
# target above) -- deliberately NOT added to IMAGE_PRG_TARGETS, since it
# isn't a real app.
set(CA65_APP_SMOKETEST_TARGET "")
if(Ca65_FOUND)
    add_ca65_app(ca65_app_smoketest
        "${CMAKE_SOURCE_DIR}/tests/src/hello/hello.s"
        CA65_APP_SMOKETEST_SRCS
        1000
        "0700"
    )
    set(CA65_APP_SMOKETEST_TARGET ca65_app_smoketest)
endif()

# 5. Test PRGs (one per tests/src/<name>/<name>.asm, automatically registered via glob with version tracking)
file(GLOB TEST_SRCS "tests/src/*/*.asm")
set(TEST_TARGETS "")
foreach(TEST_SRC ${TEST_SRCS})
    get_filename_component(TEST_NAME "${TEST_SRC}" NAME_WE)
    set(TEST_TARGET "test_${TEST_NAME}")
    
    # Create sources variable dynamically for the external app macro
    set(TMP_SRCS_${TEST_NAME} "${TEST_SRC}")
    
    # Register as external application with version/build tracking
    add_external_app(${TEST_TARGET} "${TEST_SRC}" TMP_SRCS_${TEST_NAME} 1000)
    
    list(APPEND TEST_TARGETS ${TEST_TARGET})
endforeach()

# 5.5 Automated Documentation Synchronization
# Copies documentation files from wiki/ to docs/ automatically during builds.
set(SYNC_FILES
    api-reference.md
    debug-test-plan.md
    pet-sci-api.md
    programmers-reference.md
    user-manual.md
    vmm-api.md
)

set(SYNC_OUTPUTS "")
foreach(FILE ${SYNC_FILES})
    add_custom_command(
        OUTPUT "${CMAKE_SOURCE_DIR}/docs/${FILE}"
        COMMAND "${CMAKE_COMMAND}" -E copy
            "${CMAKE_SOURCE_DIR}/wiki/${FILE}"
            "${CMAKE_SOURCE_DIR}/docs/${FILE}"
        DEPENDS "${CMAKE_SOURCE_DIR}/wiki/${FILE}"
        COMMENT "Syncing ${FILE} from wiki/ to docs/"
    )
    list(APPEND SYNC_OUTPUTS "${CMAKE_SOURCE_DIR}/docs/${FILE}")
endforeach()

add_custom_target(sync_docs ALL DEPENDS ${SYNC_OUTPUTS})

# 6. Release Target
# Gather built binary paths to pass to release script
get_target_property(CMD64_PATH command64 C64_PRG_PATH)
get_target_property(DEBUG_PATH debug C64_PRG_PATH)
set(PRG_PATHS_LIST "${CMD64_PATH};${DEBUG_PATH}")

foreach(C_TARGET ${C_TARGETS})
    get_target_property(C_PATH ${C_TARGET} C64_PRG_PATH)
    list(APPEND PRG_PATHS_LIST "${C_PATH}")
endforeach()

string(REPLACE ";" "," PRG_PATHS_STR "${PRG_PATHS_LIST}")

add_custom_target(release
    COMMAND "${CMAKE_COMMAND}"
        -DVERSION="${OS_VERSION}"
        -DSOURCE_DIR="${CMAKE_SOURCE_DIR}"
        -DBINARY_DIR="${CMAKE_BINARY_DIR}"
        -DRELEASE_DIR="${CMAKE_SOURCE_DIR}/release"
        -DPRG_PATHS="${PRG_PATHS_STR}"
        -P "${CMAKE_SOURCE_DIR}/cmake/PackRelease.cmake"
    DEPENDS image_d64 test_image_d64
    COMMENT "Packaging command64 OS v${OS_VERSION} release archives"
)
